SafeLibs Apt Repository

Memory-safe drop-in packages for selected C libraries, published as a signed static apt repository for Ubuntu 24.04. 

sudo install -d -m 0755 /etc/apt/keyrings /etc/apt/preferences.d
curl -fsSL {base_url}/{key_name}.gpg | sudo tee /etc/apt/keyrings/{key_name}.gpg > /dev/null
curl -fsSL {base_url}/{key_name}.pref | sudo tee /etc/apt/preferences.d/{key_name}.pref > /dev/null
echo "deb [signed-by=/etc/apt/keyrings/{key_name}.gpg] {base_url} {suite} {component}" | sudo tee /etc/apt/sources.list.d/{key_name}.list
sudo apt-get update
